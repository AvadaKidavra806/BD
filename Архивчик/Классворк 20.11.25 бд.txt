Датарождения date not null)

(ИДПаспортклиентa int identity (1,1) primary key,

Серия char (4 )not null,
Номер char (6) not null,
ДатаВыдачи date not null)

create table Поездка
(ИДПоездка int identity (1,1) primary key,

ИДКлиента int foreign key references Клиент (ИДКлиента) not null,
